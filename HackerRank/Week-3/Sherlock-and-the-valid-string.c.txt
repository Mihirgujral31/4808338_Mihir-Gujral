#include <stdio.h>
#include <string.h>

int main() {
    char s[100005];
    scanf("%s", s);

    int freq[26] = {0};

    for (int i = 0; s[i] != '\0'; i++) {
        freq[s[i] - 'a']++;
    }

    int counts[26], idx = 0;
    for (int i = 0; i < 26; i++) {
        if (freq[i] > 0) {
            counts[idx++] = freq[i];
        }
    }

    int min = counts[0], max = counts[0];
    for (int i = 1; i < idx; i++) {
        if (counts[i] < min) min = counts[i];
        if (counts[i] > max) max = counts[i];
    }

    int minCount = 0, maxCount = 0;
    for (int i = 0; i < idx; i++) {
        if (counts[i] == min) minCount++;
        else if (counts[i] == max) maxCount++;
    }

    if (min == max) {
        printf("YES\n"); 
    } 
    else if (minCount == 1 && min == 1 && maxCount + minCount == idx) {
        printf("YES\n"); 
    } 
    else if (maxCount == 1 && max - min == 1 && minCount + maxCount == idx) {
        printf("YES\n"); 
    } 
    else {
        printf("NO\n");
    }

    return 0;
}
