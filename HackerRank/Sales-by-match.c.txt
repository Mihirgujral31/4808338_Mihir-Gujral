#include <stdio.h>

int main() {
    int n;
    scanf("%d", &n);   

    int arr[n];
    for (int i=0; i<n; i++) {
        scanf("%d", &arr[i]);   
    }

    int freq[101] = {0};   
    for (int i=0; i<n; i++) {
        freq[arr[i]]++;
    }

    int pairs = 0;
    for (int i=0; i<101; i++) {
        pairs += freq[i]/2;   
    }

    printf("%d\n", pairs);

    return 0;
}
