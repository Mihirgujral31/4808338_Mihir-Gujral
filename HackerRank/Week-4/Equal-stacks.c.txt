#include <stdio.h>
#include <stdlib.h>

int arraySum(int *arr, int n) {
    int sum = 0;
    for(int i = 0; i < n; i++)
        sum += arr[i];
    return sum;
}

int main() {
    int n1, n2, n3;
    scanf("%d %d %d", &n1, &n2, &n3);

    int h1[n1], h2[n2], h3[n3];

    for(int i = 0; i < n1; i++) scanf("%d", &h1[i]);
    for(int i = 0; i < n2; i++) scanf("%d", &h2[i]);
    for(int i = 0; i < n3; i++) scanf("%d", &h3[i]);

    int i1 = 0, i2 = 0, i3 = 0;

    int sum1 = arraySum(h1, n1);
    int sum2 = arraySum(h2, n2);
    int sum3 = arraySum(h3, n3);

    while(!(sum1 == sum2 && sum2 == sum3)) {
        if(sum1 >= sum2 && sum1 >= sum3) {
            sum1 -= h1[i1++];
        } else if(sum2 >= sum1 && sum2 >= sum3) {
            sum2 -= h2[i2++];
        } else {
            sum3 -= h3[i3++];
        }
        if(i1 == n1 || i2 == n2 || i3 == n3) {
            printf("0\n");
            return 0;
        }
    }

    printf("%d\n", sum1);
    return 0;
}
