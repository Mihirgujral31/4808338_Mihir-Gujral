#include <stdio.h>
#include <stdlib.h>

long long arr[1000005];  

int main() {
    int n, m;
    scanf("%d %d", &n, &m);

    for (int i = 0; i <= n; i++) arr[i] = 0;

    for (int i = 0; i < m; i++) {
        int a, b;
        long long k;
        scanf("%d %d %lld", &a, &b, &k);
        arr[a] += k;
        if (b + 1 <= n) arr[b + 1] -= k;
    }

    long long maxVal = 0, curr = 0;
    for (int i = 1; i <= n; i++) {
        curr += arr[i];
        if (curr > maxVal) maxVal = curr;
    }

    printf("%lld\n", maxVal);
    return 0;
}
