#include <stdio.h>
#include <stdlib.h>

int cmp(const void *a, const void *b) {
    int x = *(int*)a, y = *(int*)b;
    return (x - y);
}

int main() {
    int n, k;
    scanf("%d %d", &n, &k);
    
    int houses[n];
    for (int i = 0; i < n; i++) scanf("%d", &houses[i]);
    
    qsort(houses, n, sizeof(int), cmp);
    
    int i = 0, transmitters = 0;
    while (i < n) {
        transmitters++;
        
        int loc = houses[i] + k;
        while (i < n && houses[i] <= loc) i++;
        
        int pos = houses[i-1];
        
        loc = pos + k;
        while (i < n && houses[i] <= loc) i++;
    }
    
    printf("%d\n", transmitters);
    return 0;
}
