#include <stdio.h>
#include <stdlib.h>

#define MAXN 100000

int arr[MAXN];
int deque[MAXN]; 

int main() {
    int n, q;
    scanf("%d %d", &n, &q);

    for (int i = 0; i < n; i++) scanf("%d", &arr[i]);

    while (q--) {
        int d;
        scanf("%d", &d);

        int front = 0, back = 0;
        int result = 1000000000;

        for (int i = 0; i < n; i++) {
            while (front < back && deque[front] <= i - d) front++;

            while (front < back && arr[deque[back - 1]] <= arr[i]) back--;

            deque[back++] = i;

            if (i >= d - 1) {
                int windowMax = arr[deque[front]];
                if (windowMax < result)
                    result = windowMax;
            }
        }

        printf("%d\n", result);
    }

    return 0;
}
