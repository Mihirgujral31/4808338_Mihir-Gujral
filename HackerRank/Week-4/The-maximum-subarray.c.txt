
#include <stdio.h>
#include <limits.h>

int max(int a, int b) {
    return (a > b) ? a : b;
}

int main() {
    int t;
    scanf("%d", &t);

    while (t--) {
        int n;
        scanf("%d", &n);
        int arr[n];

        for (int i = 0; i < n; i++) {
            scanf("%d", &arr[i]);
        }

        int current_sum = arr[0], max_subarray = arr[0];
        for (int i = 1; i < n; i++) {
            current_sum = max(arr[i], current_sum + arr[i]);
            max_subarray = max(max_subarray, current_sum);
        }

        int max_subsequence = 0;
        int all_negative = 1;
        int max_element = arr[0];

        for (int i = 0; i < n; i++) {
            if (arr[i] > 0) {
                max_subsequence += arr[i];
                all_negative = 0;
            }
            if (arr[i] > max_element) {
                max_element = arr[i];
            }
        }

        if (all_negative) {
            max_subsequence = max_element;
        }

        printf("%d %d\n", max_subarray, max_subsequence);
    }

    return 0;
}
