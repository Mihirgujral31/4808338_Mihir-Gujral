#include <stdio.h>
#include <stdlib.h>

int isLeap(int y) {
    if ((y%400 == 0) || (y%4 == 0 && y%100 != 0))
        return 1;
    return 0;
}

int daysInMonth(int m, int y) {
    switch (m) {
        case 1:  return 31;
        case 2:  return isLeap(y) ? 29 : 28;
        case 3:  return 31;
        case 4:  return 30;
        case 5:  return 31;
        case 6:  return 30;
        case 7:  return 31;
        case 8:  return 31;
        case 9:  return 30;
        case 10: return 31;
        case 11: return 30;
        case 12: return 31;
        default: return 0;
    }
}

void nextDate(int *d, int *m, int *y) {
    int dim = daysInMonth(*m, *y);
    if (*d < dim) {
        (*d)++;
    }
    else {
        *d = 1;
        if (*m < 12) {
            (*m)++;
        }
        else {
            *m = 1;
            (*y)++;
        }
    }
}

int main() {
    char u[15], v[15];
    scanf("%s %s", u, v);

    int d1, m1, y1, d2, m2, y2;
    sscanf(u, "%d-%d-%d", &d1, &m1, &y1);
    sscanf(v, "%d-%d-%d", &d2, &m2, &y2);

    long long count = 0;
    int d = d1, m = m1, y = y1;

    while (y<y2 || (y==y2 && (m<m2 || (m==m2 && d<=d2)))) {
        long long N = (d * 100 + m) + y;
        if (N%4 == 0 || N%7 == 0) {
            count++;
        }

        if (d == d2 && m == m2 && y == y2) {
            break;
        }
        nextDate(&d, &m, &y);
    }
    printf("%lld\n", count);
    return 0;
}
