#include <stdio.h>

int main() {
    int n;
    scanf("%d", &n);
    
    float pos, neg, zero = 0;
    
    int arr[n];
    for(int i=0; i<n; i++) {
        scanf("%d", &arr[i]);
        
        if(arr[i] > 0) {
            pos++;
        }
        else if(arr[i] < 0) {
            neg++;
        }
        else {
            zero++;
        }
    }
    printf("%f\n", pos/n);
    printf("%f\n", neg/n);
    printf("%f\n", zero/n);
}
