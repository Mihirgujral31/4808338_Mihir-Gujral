#include <stdio.h>



int main() {
    int arr[5], size=5;
    for(int i=0; i<5; i++) {
        scanf("%d", &arr[i]);
    }
    
    for(int i=0; i<size-1; i++) {
        for(int j=0; j<size-i-1; j++) {
            if(arr[j] > arr[j+1]) {
                int temp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = temp;
            }
        }
    }
    
    long long min_sum = 0;
    
    for(int i=0; i<size-1; i++) {
        min_sum += arr[i];
    }
    printf("%lld ", min_sum);
    
    long long max_sum = 0;
    for(int i=1; i<size; i++) {
        max_sum += arr[i];
    }
    printf("%lld", max_sum);
}
