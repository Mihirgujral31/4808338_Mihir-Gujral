
#include <stdio.h>
#include <stdlib.h>

int main() {
    int n;
    scanf("%d", &n);
    int arr[n][n];
    for(int i=0; i<n; i++) {
        for(int j=0; j<n; j++) {
            scanf("%d", &arr[i][j]);
        }
    }
    
    int left_sum = 0, right_sum = 0;
    for(int i=0; i<n; i++) {
        left_sum = left_sum + arr[i][i];
        right_sum = right_sum + arr[i][n-i-1];
    }
    int diff = abs(left_sum - right_sum);
    printf("%d", diff);
}
