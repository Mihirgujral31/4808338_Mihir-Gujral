#include <stdio.h>

int main() {
    char s[11]; 
    scanf("%s", s);

    int hour, minute, second;
    char period[3];

    sscanf(s, "%2d:%2d:%2d%2s", &hour, &minute, &second, period);

    if(period[0]=='A' && hour==12) {
        hour=0; 
    } else if(period[0]=='P' && hour!=12) {
        hour+=12; 
    }

    printf("%02d:%02d:%02d\n", hour, minute, second);

}
